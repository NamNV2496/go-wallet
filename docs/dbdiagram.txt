// Docs: https://dbml.dbdiagram.io/docs

Table accounts {
  ID bigint [primary key]
  user_id bigint [ref: - users.id]
  Balance bigint
  Currency varchar(3)
  created_at timestamp 
}

Table transfers {
  id bigint [primary key]
  from_account_id bigint [ref: - users.id]
  to_account_id bigint [ref: - users.id]
  amount bigint
  status integer
  message varchar
  created_at timestamp
}

Table users {
  id bigint [primary key]
  username varchar unique
  hashed_password varchar
  full_name varchar
  email varchar unique
  phone varchar
  is_email_verified varchar unique
  password_changed_at timestamp
  created_at timestamp
}
